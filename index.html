Sub CrearLibroFinanciero_Excel() Dim wb As Workbook: Set wb = ThisWorkbook Dim ws As Worksheet ' Registro_Movimientos On Error Resume Next Application.DisplayAlerts = False wb.Worksheets("Registro_Movimientos").Delete wb.Worksheets("Balance").Delete wb.Worksheets("Inversiones_Ahorro").Delete wb.Worksheets("Config_Categorias").Delete Application.DisplayAlerts = True On Error GoTo 0 Set ws = wb.Worksheets.Add(After:=wb.Sheets(wb.Sheets.Count)) ws.Name = "Registro_Movimientos" ws.Range("A1:H1").Value = Array("ID","Fecha","Concepto","Categoria","Pilar_Regla","Monto","Cuenta","Usuario") ws.Range("A2:H2").Value = Array(1, "2026-04-19", "Carga Inicial", "Guardia", "Ingreso", 0, "Banco", "Admin") ws.Columns("B").NumberFormat = "yyyy-mm-dd" ws.Columns("F").NumberFormat = "$#,##0.00" ' Balance Set ws = wb.Worksheets.Add(After:=wb.Sheets(wb.Sheets.Count)) ws.Name = "Balance" ws.Range("A1:B1").Value = Array("Pilar","Monto") ws.Range("A2").Value = "Necesidades (50%)" ws.Range("B2").Formula = "=SUMIF(Registro_Movimientos!E:E,""Necesidades (50%)"",Registro_Movimientos!F:F)" ws.Range("A3").Value = "Deseos (30%)" ws.Range("B3").Formula = "=SUMIF(Registro_Movimientos!E:E,""Deseos (30%)"",Registro_Movimientos!F:F)" ws.Range("A4").Value = "Ahorro/Inversion (20%)" ws.Range("B4").Formula = "=SUMIF(Registro_Movimientos!E:E,""Ahorro/Inversion (20%)"",Registro_Movimientos!F:F)" ' Inversiones_Ahorro Set ws = wb.Worksheets.Add(After:=wb.Sheets(wb.Sheets.Count)) ws.Name = "Inversiones_Ahorro" ws.Range("A1:E1").Value = Array("Activo","Ticker_Ref","Cantidad","Precio_Actual","Valor_Total_ARS") ws.Range("A2:E4").Value = Array( _ Array("Apple","NASDAQ:AAPL",0,"", "=C2*D2"), _ Array("Nvidia","NASDAQ:NVDA",0,"", "=C3*D3"), _ Array("Metroterra","Manual",1,0,"=C4*D4") _ ) ' Config_Categorias Set ws = wb.Worksheets.Add(After:=wb.Sheets(wb.Sheets.Count)) ws.Name = "Config_Categorias" ws.Range("A1:B1").Value = Array("Categoria","Pilar_Asociado") ws.Range("A2:B6").Value = Array( _ Array("Guardia","Ingreso"), _ Array("Alquiler","Necesidades"), _ Array("Supermercado","Necesidades"), _ Array("Salidas","Deseos"), _ Array("Inversión","Ahorro") _ ) MsgBox "Estructura creada. Completa precios en Inversiones_Ahorro y ajusta estilos manualmente.", vbInformation End Sub
